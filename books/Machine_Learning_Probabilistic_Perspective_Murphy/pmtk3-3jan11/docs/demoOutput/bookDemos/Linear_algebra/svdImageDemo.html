loadData('clown');
r = rank(X);
disp(r);
[U, S, V] = svd(X, 0);
ranks = [1 2 5 10 20 r];
R = length(ranks);

   200

for i=1:R
  figure();
  k = ranks(i);
  Xhat = (U(:, 1:k)*S(1:k, 1:k)*V(:, 1:k)');
  image(Xhat);
  colormap(gray(64));
  axis image;
  axis off
  title(sprintf('rank %d', k))
  printPmtkFigure(sprintf('svdImageDemoClown%d', k));
end

sigma = diag(S);
K = 100;
figure();
plot(log(sigma(1:K)), 'r-', 'linewidth', 3)
ylabel(sprintf('log(%s_i)','\sigma'));
xlabel('i')
printPmtkFigure svdImageDemoClownSigma;
